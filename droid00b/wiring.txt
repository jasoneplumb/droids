AirLift -> RP2040 Feather
VIN -> (open)
3Vo -> 3.3V
GND -> GND
SCK -> SCK
MISO -> MI
MOSI -> MO
CS -> 10
BUSY -> 9
RST -> 6
GP0 -> (open)
RXI -> (open)
TXO -> (open)

Soil Sensor -> RP2040 Feather
Black -> GND
Red -> 3.3V
White -> A0 (w/ 10kOhm pull up)
Green -> A1 (w/ 10kOhm pull up)

UV Sensor -> RP2040 Feather
- -> GND
Out -> A3
+ -> 3.3V

BME280 Sensor -> RP2040 Feather
VIN -> (open)
3Vo -> 3.3V
GND -> GND
SCK -> SCL
SDO -> (open)
SDI -> SDA
